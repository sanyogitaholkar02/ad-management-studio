CREATE TABLE IF NOT EXISTS `ads-agg` (
    id          UInt64,
    window_start DateTime64(3),
    adId        String,
    campaignId  String,
    clicks      UInt64,
    revenue     Float64
) ENGINE = MergeTree()
ORDER BY (window_start, adId, campaignId);


CREATE TABLE IF NOT EXISTS `user_click_rate` (
    userId       String,
    window_start DateTime64(3),
    window_end   DateTime64(3),
    clicks       UInt64,
    total_cost   Float64
) ENGINE = MergeTree()
ORDER BY (window_start, userId);

-- User click rate: 24h rollup (this is what the API does)
SELECT
    userId,
    SUM(clicks) AS total_clicks,
    SUM(total_cost) AS total_cost
FROM user_click_rate
WHERE window_start >= now() - INTERVAL 24 HOUR
GROUP BY userId
ORDER BY total_clicks DESC;
